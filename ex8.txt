#include <stdio.h>

int main() {
    float m;
    printf("Entrez la moyenne : ");
    scanf("%f", &m);

    if (m < 10)
        printf("Échec.\n");
    else if (m < 12)
        printf("Passable.\n");
    else if (m < 14)
        printf("Assez Bien.\n");
    else if (m < 16)
        printf("Bien.\n");
    else
        printf("Très Bien.\n");

    return 0;
}
