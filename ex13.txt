#include <stdio.h>

int main() {
    int x1, y1, x2, y2, x, y;
    printf("Entrer les coordonnées du segment (x1 y1 x2 y2) : ");
    scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
    printf("Entrer le point (x y) : ");
    scanf("%d %d", &x, &y);

    // Vérifie si le point est aligné et dans l'intervalle
    if ((x-x1)*(y2-y1) == (y-y1)*(x2-x1) &&
        x >= (x1<x2?x1:x2) && x <= (x1>x2?x1:x2) &&
        y >= (y1<y2?y1:y2) && y <= (y1>y2?y1:y2))
        printf("Le point appartient au segment.\n");
    else
        printf("Le point n'appartient pas au segment.\n");

    return 0;
}
