#include <stdio.h>

int main() {
    int h1, m1, s1, h2, m2, s2;
    printf("Entrez l'heure 1 (HH MM SS) : ");
    scanf("%d %d %d", &h1, &m1, &s1);
    printf("Entrez l'heure 2 (HH MM SS) : ");
    scanf("%d %d %d", &h2, &m2, &s2);

    if (h1 < h2 || (h1==h2 && m1<m2) || (h1==h2 && m1==m2 && s1<s2))
        printf("La première précède la deuxième.\n");
    else if (h1==h2 && m1==m2 && s1==s2)
        printf("Les deux sont identiques.\n");
    else
        printf("La deuxième précède la première.\n");

    return 0;
}
