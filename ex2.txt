#include <stdio.h>

int main() {
    char c;
    printf("Entrez une lettre : ");
    scanf(" %c", &c);

    switch(c) {
        case 'a': case 'A':
        case 'e': case 'E':
        case 'i': case 'I':
        case 'o': case 'O':
        case 'u': case 'U':
            printf("%c est une voyelle.\n", c);
            break;
        default:
            printf("%c n'est pas une voyelle.\n", c);
    }
    return 0;
}
